43std::vector<double> Polynomial::TimeInterval(double start_in, double end_in, int n)
44{
45
46 std::vector<double> linspaced;
47
48 double start = static_cast<double>(start_in);
49 double end = static_cast<double>(end_in);
50 double num = static_cast<double>(n);
51
52 if (num == 0)
53 {
54 return linspaced;
55 }
56 if (num == 1)
57 {
58 linspaced.push_back(start);
59 return linspaced;
60 }
61
62 double delta = (end - start) / (num - 1);
63
64 for (int i = 0; i < num - 1; ++i)
65 {
66 linspaced.push_back(start + delta * i);
67 }
68 linspaced.push_back(end);
69 return linspaced;
70}

160std::vector<double> Polynomial::PolynomialPointsFromCoefficients(std::vector<double> coefficients, double from_t, double to_t, int n)
161{
162
163#ifdef DEBUG2
164 std::cout << "Computing [ " << n << " ] Uniform points between [ " << from_t << " , " << to_t << " ]" << std::endl;
165#endif
166 std::vector<double> ts = TimeInterval(from_t, to_t, n);
167 std::vector<double> polynomial;
168 polynomial.resize(n);
169 for (auto &p : polynomial)
170 {
171 p = 0;
172 }
173#ifdef DEBUG1
174 std::cout << " Point Intervals: " << polynomial.size() << std::endl;
175 std::cout << " [ ";
176 for (const auto &t : ts)
177 {
178 std::cout << t << " , ";
179 }
180 std::cout << " ] " << std::endl;
181#endif
182#ifdef DEBUG2
183 std::cout << "Computing polynomials: " << std::endl;
184#endif
185 int power{0};
186 for (size_t t{0}; t < ts.size(); ++t)
187 {
188 for (const auto &coefficient : coefficients)
189 {
190 polynomial[t] = polynomial[t] + (coefficient * std::pow(ts[t], power));
191 ++power;
192 }
193 power = 0;
194 }
195#ifdef DEBUG2
196 std::cout << "Computed polynomials." << std::endl;
197#endif
198#ifdef DEBUG1
199 std::cout << " Polynomial vector size is : " << polynomial.size() << std::endl;
200 std::cout << " [ ";
201 for (const auto &p : polynomial)
202 {
203 std::cout << p << " , ";
204 }
205 std::cout << " ] " << std::endl
206 << std::endl;
207#endif
208
209 return polynomial;
210}

219std::vector<double> Polynomial::ComputeVelocity(std::vector<double> coefficient, std::vector<double> time)
220{
221 std::vector<double> velocity;
222 for (size_t t{0}; t < time.size(); ++t)
223 {
224 velocity.push_back(coefficient[1] + 2 * coefficient[2] * time[t] +
225 3 * coefficient[3] * std::pow(time[t], 2.0) +
226 4 * coefficient[4] * std::pow(time[t], 3.0)+
227 5 * coefficient[5] * std::pow(time[t], 4.0));
228 }
229 return velocity;
230}
